40 static Mat paramsToProjection(const Mat & p,const Mat& _cameraMatrix){
41  //Build the base transform
42  assert(p.type()==CV_64FC1);
43  Mat dR=rodrigues(p.colRange(Range(0,3)));
44  Mat dT=p.colRange(Range(3,6)).t();
45  Mat dA;
46  hconcat(dR,dT,dA);
47  dA=make4x4(dA);
48  Mat cameraMatrix=make4x4(_cameraMatrix);
49  assert(cameraMatrix.type()==CV_64FC1);
50  Mat proj=cameraMatrix*dA*cameraMatrix.inv();
51 // cout<<"p: "<<"\n"<< p<< endl;
52 // cout<<"Proj: "<<"\n"<< proj<< endl;
53  //The column swap
54  Mat tmp=proj.colRange(2,4).clone();
55  tmp.col(1).copyTo(proj.col(2));
56  tmp.col(0).copyTo(proj.col(3));
57  //The row drop
58  proj=proj.rowRange(0,3);
59  return proj;
60 }

70  //Image gradients for alignment
71  //Note that these gradients have theoretical problems under the sudden
72  // changes model of images. It might be wise to blur the images before
73  // alignment, to avoid sudden changes, but that makes occlusion more
74  // problematic.
75  grad.create(2,image.rows*image.cols,CV_32FC1);
76  Mat gray;
77  if (image.type()==CV_32FC1) {
78  gray=image;
79  }else {
80  cvtColor(image, gray, CV_BGR2GRAY);
81  gray.convertTo(gray,CV_32FC1);
82  }
83  Mat grad_x(image.rows,image.cols,CV_32FC1,grad.row(0).data);
84  Scharr( gray, grad_x, CV_32FC1, 1, 0, 1.0/26.0, 0, BORDER_REPLICATE );
